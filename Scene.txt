--@include lib/Class.txt
require("lib/Class.txt")

Scene = Class("Scene")
Scene.objects = {}

function Scene:addObject(obj)
	if obj == nil then return end --TODO: Add exception
	Scene.objects[obj] = true
end

function Scene:removeObject(obj)
	if obj == nil then return end --TODO: Add exception
	Scene.objects[obj] = nil
end

function Scene:draw()
	render.clear()
	for k, _ in pairs(Scene.objects) do
		k:draw()
	end
end

function Scene:think()
	for k, _ in pairs(Scene.objects) do
		k:think()
	end
end
